    # sync

    # 1.sync

    之前父子组件传值、修改父组件的值，只能靠$emit触发自定义事件通知父组件去修改，相对来说比较繁琐

    // 父组件
<template>
  <div class="hello">
    <child :test1 = 'test1' @changeTest1='change'></child>
  </div>
</template>

<script>
import child from './child'
export default {
  name: 'HelloWorld',
  data(){
    return{
      test1: 'test1',
    }
  },
  components:{ child },
  props: { msg: String },
  methods: {
    change(val){
      this.test1 = val
    }
  }
}
</script>

//  子组件
<template>
    <div>
        <p>我是儿子</p>
        <p>test1:{{test1}}</p>
        <button @click="change">修改</button>
    </div>
</template>
<script>
export default {
    props:['test1'],
    methods:{
        change(){
            this.$emit('changeTest1',`修改过的test1`)
        }
    }
}
</script>

    2.4新增的.sync写法相对舒服多了。

    // 父组件
<template>
  <div class="hello">
    <child :test1.sync = 'test1' ></child>
  </div>
</template>

<script>
import child from './child'
export default {
  name: 'HelloWorld',
  data(){
    return{
      test1: 'test1',
    }
  },
  components:{ child },
}
</script>

// 子组件
<template>
    <div>
        <p>我是儿子</p>
        <p>test1:{{test1}}</p>
        <button @click="change">修改</button>
    </div>
</template>
<script>
export default {
    props:['test1'],
    methods:{
        change(){
            this.$emit('update:test1',`修改过的test1`)
        }
    }
}
</script>
