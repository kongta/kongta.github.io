    # 自定义指令

    # 1.生命周期

    bind() {} // 被绑定
inserted() {} // 绑定到节点
update() {} // 组件更新（指令的值可能发生了改变，也可能没有。）
componentUpdate() {} // 组件更新完成（指令所在组件的 VNode 及其子 VNode 全部更新后调用。）
unbind() {} // 解绑

    # 2.自定义指令传递的三个参数

    el: 指令所绑定的元素，可以用来直接操作DOM。
binding: 一个对象，包含指令的很多信息。
binding包括：name：指令名，不包括 v- 前缀。
    		value：指令的绑定值，例如：v-my-directive="1 + 1" 中，绑定值为 2。
            oldValue：指令绑定的前一个值，仅在 update 和 componentUpdated 钩子中可用。无论值是否改			变都可用。
    		expression：字符串形式的指令表达式。例如 v-my-directive="1 + 1" 中，表达式为 "1 + 			 1"。
    		arg：传给指令的参数，可选。例如 v-my-directive:foo 中，参数为 "foo"。
    		modifiers：一个包含修饰符的对象。例如：v-my-directive.foo.bar 中，修饰符对象为 { foo: 			   true, 	 bar: true }。
vnode: Vue编译生成的虚拟节点。

    # 3.用法

    src目录下建一个directive文件夹，建一个color.js文件

    // color.js
const color = {
  inserted: function(el) {
    el.style.color = "green";
  }
};
export default color;

    建一个index.js文件

    // index.js
import color from "./color";
export default {
  color
};

    在main.js中注册引用

    // main.js
import directives from "./directive";
Object.keys(directives).forEach(directiveName => {
  Vue.directive(directiveName, directives[directiveName]);
});

    在组件中使用

    <p v-color>自定义指令</p>

    # 实现一个复制的指令

    // copy.js
const copy = {
  bind(el, { value }) {
    el.$value = value; //其他生命周期可用
    el.handler = () => {
      if (!value) {
        return alert("无复制内容");
      }
      const textarea = document.createElement("textarea");
      textarea.readOnly = "readonly"; // 防止唤起键盘
      textarea.style.position = "absolute";
      textarea.style.left = "-99999px";
      textarea.value = el.$value;
      document.body.append(textarea);
      textarea.select();
      textarea.setSelectionRange(0, textarea.value.length);
      const result = document.execCommand("Copy");
      if (result) {
        alert("复制成功");
      }
      document.body.removeChild(textarea);
    };
    el.addEventListener("click", el.handler);
  },
  componentUpdated(el, { value }) {
    el.$value = value;
  },
  unbind(el) {
    el.removeEventListener("click", el.handler);
  }
};

export default copy;

    组件中使用

    <template>
  <div id="app">
      <button v-copy="message">复制</button>
  </div>
</template>
<script>
export default {
  data() {
    return {
      message: "你好吗"
    };
  },
};
</script>
