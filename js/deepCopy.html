    # 深拷贝

    一开始使用的深拷贝

        let obj = { a: 111 }
    let obj2 = JSON.parse(JSON.stringify(obj))
    obj2.a = 222
    obj // {a: 111}
    ob2 // {a: 222}

    哇！！！深拷贝也不过如此

        let arr = [1, 2, 3, 4, undefined]
    let arr2 = JSON.parse(JSON.stringify(obj))
    arr  // [1, 2, 3, 4, undefind]

    这个方法对象里面有方法，或者应用自身的值，抑或有undefind都会出问题。

    主菜----深拷贝终极版

    const deepCopy = obj =>{
    if (typeof obj !== 'object') return obj
    let newObj = Array.isArray(obj) ?  [] : {} 
    for (let key in obj) {
        newObj[key] = typeof obj[key] === 'object' ? deepCopy(obj[key]) : obj[key]
    }
    return newObj
}

    // 普通数组
    let arr = [1,2,3,4,5] 
    let arr2 = deepCopy(arr)
    arr2[0] = 9
    arr // [1,2,3,4,5]
    arr2 // [9,2,3,4,5]

    // 普通对象
    let obj = {a: 111, b: 222}
    let obj2 = deepCopy(obj)
    obj2.a = 333
    obj  // {a: 333, b: 222}
    obj2 // {a: 111, b: 222}

    // 多为数组
    let arr = [1, 2, 3, 4, {a:{b:5}}]
    let arr2 = deepCopy(arr)
    arr2[4].a.b = 6
    arr //  [1, 2, 3, 4, {a:{b:5}}]
    arr2 //  [1, 2, 3, 4, {a:{b:6}}]

    // 含有undefind
    let arr = [1, 2, 3, 4, undefined]
    let arr2 = deepCopy(arr)
    arr2[0] = 9
    arr // [1, 2, 3, 4, undefined]
    arr2 // [9, 2, 3, 4, undefined]
