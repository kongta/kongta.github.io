    # Reduce用法

    # 1.求和

    let arr = [1,2,3,4,5]
let sum = arr.reduce((pre, cur) => pre + cur) // 15

    # 2.求积

    let arr = [1,2,3,4,5]
let mul = arr.reduce((pre, cur) => pre * cur) // 120

    # 3.统计数组中每个元素出现的次数

    let arr = ['A','A','B','B','C','D','A']
let countObj = arr.reduce((pre, cur) => {
   	if (cur in pre) {
        pre[cur] ++
    } else {
        pre[cur] = 1
    }
}, {})
countObj // {A: 3, B: 2, C: 1, D: 1}

    # 4.数组去重

    let arr = ['A','A','B','B','C','D','A']
let newArr = arr.reduce((pre, cur) => {
    if(pre.includes(cur)) {
        return pre
    } else {
        return [...pre,cur]
    }
}, [])
newArr // ["A", "B", "C", "D"]

    # 5.数组扁平化

    // 二维数组
let arr = [[1,2],[3,4],[5,6]]
let newArr = arr.reduce((pre, cur)=> [...pre,...cur],[]) // [1,2,3,4,5,6]

// 多维数组
let arr2 = [[1,2,[3,4,[5,6]]],[7,8,[10,11]]]
const flatArrFun = arr => {
   return arr.reduce((pre, cur) => pre.concat(Array.isArray(cur) ? flatArrFun(cur) : cur),[])
}
flatArrFun(arr2) // [1, 2, 3, 4, 5, 6, 7, 8, 10, 11]
