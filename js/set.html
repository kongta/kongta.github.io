    # Set

    # es6新的数据结构，类似于数组，所有的值都是唯一的，不可以重复

    # Set是一个构造函数，接受一个数组（或者具有 iterable 接口的其他数据结构）来初始化

    # 属性

    let s = new Set([1,2,3,3,3,4,5]) // Set(5) {1, 2, 3, 4, 5}
s.size // 5

    # 操作方法

    let s = new Set([1,2,3,3,3,4,5]) 
s.has(5) // true
s.delete(5) // true  s => Set(4) {1, 2, 4, 5}
s.add(6) // Set(5) {1, 2, 4, 5, 6}
s.clear() // Set(0) {}

    # 运用

    数组去重

    let arr = [1, 1, 2, 2, 3, 4, 5]
let arr2 = Array.from(new Set(arr)) // [1, 2, 3, 4, 5]
let arr3 = [...new Set(arr)] // [1, 2, 3, 4, 5]

    数组交集

    let arr = [1, 2, 3, 4]
let arr2 = [2, 3, 4, 5]
arr.filter(x => new Set(arr2).has(x)) // [2, 3, 4]
// 如果arr = [1, 2, 3, 4] 上面的结果为 [2, 2, 3, 4]
Array.from(new Set(arr.filter(x=>new Set(arr2).has(x)))) // [2, 3, 4]

    数组并集

    let arr = [1, 2, 3, 4]
let arr2 = [2, 3, 4, 5]
Array.from(new Set([...new Set(arr), ...new Set(arr2)])) // [1, 2, 3, 4, 5]
    # 数组差集
    let arr = [1, 2, 3, 4]
let arr2 = [2, 3, 4, 5]
Array.from(new Set(arr.filter(x=>new Set(arr2).!has(x)))) // [2, 3, 4]
