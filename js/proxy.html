    # 要实现点击按钮number增加
      <div class="number">1</div>
  <button class="btn">增加</button>
    # 普通的操作
    document.querySelector('.btn').addEventListener('click', function(){
  let number = document.querySelector('.number')
  number.innerHTML = Number(number.innerHTML) + 1 
})
    # 利用Object.defineProperty的set方法监听属性的变化获得的值，赋予该元素，代码虽然多了但是可读性高了。
    let obj = {val :1}
    let value= 1
    Object.defineProperty(obj,'val',{
      get:function(){
      	return value
      },
      set:function(newValue){
        value = newValue
        let number = document.querySelector('.number')
        number.innerHTML = value
      }
    })
    document.querySelector('.btn').addEventListener('click', function(){
      obj.val += 1
    })
    # 再封装一个watch的方法, 灵活
    function watch(obj, prop, func){
  let value = obj[prop]
  Object.defineProperty(obj,prop,{
    get:function() {
      return value
    },
    set:function(newValue) {
      value = newValue
      func(newValue)
    }
  })
}
let obj = {value: 1}
watch(obj, 'value', function(newValue){
  document.querySelector('.number').innerHTML = newValue
})
document.querySelector('.btn').addEventListener('click', function(){
  obj.value += 1
})
